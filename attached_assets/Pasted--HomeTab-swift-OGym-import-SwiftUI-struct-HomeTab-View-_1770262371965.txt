import SwiftUI

struct HomeTab: View {
    @EnvironmentObject var authViewModel: AuthViewModel
    @State private var workoutSummary: WorkoutSummary?
    @State private var todayWorkout: TodayWorkout?
    @State private var ownerStats: OwnerStats?
    @State private var trainerStats: TrainerStats?
    @State private var isLoading = true
    @State private var showActiveWorkout = false
    
    var body: some View {
        NavigationStack {
            ScrollView {
                VStack(spacing: 20) {
                    // Welcome Header
                    HStack {
                        VStack(alignment: .leading, spacing: 4) {
                            Text(greetingText)
                                .font(.subheadline)
                                .foregroundColor(.secondary)
                            Text(authViewModel.user?.username ?? "User")
                                .font(.title2.bold())
                        }
                        Spacer()
                        if let role = authViewModel.user?.role {
                            Text(role.rawValue.capitalized)
                                .font(.caption)
                                .padding(.horizontal, 12)
                                .padding(.vertical, 6)
                                .background(roleColor(role))
                                .foregroundColor(.white)
                                .cornerRadius(12)
                        }
                    }
                    .padding(.horizontal)
                    
                    // Role-Specific Dashboard
                    switch authViewModel.user?.role {
                    case .owner:
                        OwnerDashboardContent(stats: ownerStats, isLoading: isLoading)
                    case .trainer:
                        TrainerDashboardContent(stats: trainerStats, isLoading: isLoading)
                    case .member:
                        MemberDashboardContent(
                            workoutSummary: workoutSummary,
                            todayWorkout: todayWorkout,
                            isLoading: isLoading,
                            onStartWorkout: { showActiveWorkout = true }
                        )
                    default:
                        MemberDashboardContent(
                            workoutSummary: workoutSummary,
                            todayWorkout: todayWorkout,
                            isLoading: isLoading,
                            onStartWorkout: { showActiveWorkout = true }
                        )
                    }
                    
                    Spacer(minLength: 50)
                }
                .padding(.vertical)
            }
            .navigationBarHidden(true)
            .refreshable { await fetchData() }
            .onAppear { Task { await fetchData() } }
            .fullScreenCover(isPresented: $showActiveWorkout) {
                if let workout = todayWorkout, let items = workout.items {
                    ActiveWorkoutView(
                        dayLabel: workout.dayLabel ?? "Workout",
                        exercises: items,
                        onComplete: {
                            showActiveWorkout = false
                            Task { await fetchData() }
                        }
                    )
                }
            }
        }
    }
    
    func roleColor(_ role: UserRole) -> Color {
        switch role {
        case .owner: return .purple
        case .trainer: return .orange
        case .member: return .indigo
        }
    }
    
    var greetingText: String {
        let hour = Calendar.current.component(.hour, from: Date())
        if hour < 12 { return "Good morning," }
        else if hour < 17 { return "Good afternoon," }
        else { return "Good evening," }
    }
    
    func fetchData() async {
        isLoading = true
        let role = authViewModel.user?.role
        
        if role == .member || role == nil {
            // Member endpoints
            if let url = URL(string: "https://app.ogym.fitness/api/workouts/today") {
                let request = AuthViewModel.createAuthenticatedRequest(url: url)
                if let (data, _) = try? await URLSession.shared.data(for: request),
                   let workout = try? JSONDecoder().decode(TodayWorkout.self, from: data) {
                    await MainActor.run { self.todayWorkout = workout }
                }
            }
            
            if let url = URL(string: "https://app.ogym.fitness/api/member/workout/summary") {
                let request = AuthViewModel.createAuthenticatedRequest(url: url)
                if let (data, _) = try? await URLSession.shared.data(for: request),
                   let summary = try? JSONDecoder().decode(WorkoutSummary.self, from: data) {
                    await MainActor.run { self.workoutSummary = summary }
                }
            }
        } else if role == .owner {
            // Owner dashboard endpoint
            if let url = URL(string: "https://app.ogym.fitness/api/owner/dashboard") {
                let request = AuthViewModel.createAuthenticatedRequest(url: url)
                if let (data, _) = try? await URLSession.shared.data(for: request),
                   let stats = try? JSONDecoder().decode(OwnerStats.self, from: data) {
                    await MainActor.run { self.ownerStats = stats }
                }
            }
        } else if role == .trainer {
            // Trainer dashboard endpoint
            if let url = URL(string: "https://app.ogym.fitness/api/trainer/dashboard") {
                let request = AuthViewModel.createAuthenticatedRequest(url: url)
                if let (data, _) = try? await URLSession.shared.data(for: request),
                   let stats = try? JSONDecoder().decode(TrainerStats.self, from: data) {
                    await MainActor.run { self.trainerStats = stats }
                }
            }
        }
        
        await MainActor.run { isLoading = false }
    }
}

// MARK: - Today's Workout Expandable Card
struct TodayWorkoutExpandable: View {
    let todayWorkout: TodayWorkout?
    @Binding var isExpanded: Bool
    @Binding var completedExercises: Set<Int>
    let onStartWorkout: () -> Void
    let onRefresh: () -> Void  // Add this to refresh data after marking done
    
    var exercises: [WorkoutItem] {
        todayWorkout?.items ?? []
    }
    
    var completedCount: Int {
        completedExercises.count
    }
    
    var totalCount: Int {
        exercises.count
    }
    
    var isAllCompleted: Bool {
        completedCount == totalCount && totalCount > 0
    }
    
    var body: some View {
        VStack(spacing: 0) {
            // Header
            Button(action: { withAnimation { isExpanded.toggle() } }) {
                HStack(spacing: 12) {
                    // Dumbbell icon
                    ZStack {
                        RoundedRectangle(cornerRadius: 12)
                            .fill(Color.indigo)
                            .frame(width: 44, height: 44)
                        Image(systemName: "dumbbell.fill")
                            .foregroundColor(.white)
                    }
                    
                    VStack(alignment: .leading, spacing: 2) {
                        Text("Today's Workout")
                            .font(.subheadline)
                            .foregroundColor(.secondary)
                        if let workout = todayWorkout, let label = workout.dayLabel {
                            Text(label)
                                .font(.headline)
                        } else {
                            Text("No workout scheduled")
                                .font(.headline)
                        }
                    }
                    
                    Spacer()
                    
                    if !exercises.isEmpty {
                        // Done button
                        Button(action: markAllDone) {
                            Text("Done")
                                .font(.caption.bold())
                                .padding(.horizontal, 12)
                                .padding(.vertical, 6)
                                .background(isAllCompleted ? Color.green : Color.green)
                                .foregroundColor(.white)
                                .cornerRadius(16)
                        }
                        .disabled(isAllCompleted) // Disable if already done
                        
                        Image(systemName: isExpanded ? "chevron.up" : "chevron.down")
                            .foregroundColor(.secondary)
                    }
                }
                .padding()
            }
            .buttonStyle(.plain)
            
            // Progress bar
            if !exercises.isEmpty {
                VStack(spacing: 8) {
                    HStack {
                        Text("Progress")
                            .font(.caption)
                            .foregroundColor(.secondary)
                        Spacer()
                        Text("\(completedCount)/\(totalCount)")
                            .font(.caption.bold())
                            .foregroundColor(.green)
                    }
                    
                    GeometryReader { geometry in
                        ZStack(alignment: .leading) {
                            RoundedRectangle(cornerRadius: 4)
                                .fill(Color(.systemGray5))
                                .frame(height: 8)
                            RoundedRectangle(cornerRadius: 4)
                                .fill(Color.green)
                                .frame(width: geometry.size.width * CGFloat(completedCount) / CGFloat(max(totalCount, 1)), height: 8)
                        }
                    }
                    .frame(height: 8)
                }
                .padding(.horizontal)
                .padding(.bottom, 12)
            }
            
            // Expanded exercises list
            if isExpanded && !exercises.isEmpty {
                VStack(spacing: 8) {
                    ForEach(Array(exercises.enumerated()), id: \.element.id) { index, exercise in
                        ExerciseRowCheckable(
                            exercise: exercise,
                            isCompleted: completedExercises.contains(exercise.id),
                            onToggle: { toggleExercise(exercise.id) }
                        )
                    }
                }
                .padding(.horizontal)
                .padding(.bottom, 12)
            }
        }
        .background(Color(.systemGray6))
        .cornerRadius(16)
        .padding(.horizontal)
        .onAppear {
            // Load completed state from API response
            if todayWorkout?.dayManuallyCompleted == true {
                for exercise in exercises {
                    completedExercises.insert(exercise.id)
                }
            } else {
                for exercise in exercises {
                    if exercise.completed == true {
                        completedExercises.insert(exercise.id)
                    }
                }
            }
        }
    }
    
    // MARK: - Mark All Done (calls API)
    private func markAllDone() {
        // Don't allow if already completed
        guard !isAllCompleted else { return }
        
        // Mark all locally first
        for exercise in exercises {
            completedExercises.insert(exercise.id)
        }
        
        // Call API to mark day as done
        Task {
            await markDayDoneAPI()
        }
    }
    
    // MARK: - Toggle Individual Exercise
    private func toggleExercise(_ id: Int) {
        // If already completed, don't allow unchecking (API doesn't support it)
        if completedExercises.contains(id) {
            // Already completed - cannot uncomplete
            return
        }
        
        // Mark as completed locally
        completedExercises.insert(id)
        
        // Call API to complete this exercise
        Task {
            await completeExerciseAPI(workoutItemId: id)
        }
    }
    
    // MARK: - API: Complete Single Exercise
    private func completeExerciseAPI(workoutItemId: Int) async {
        guard let url = URL(string: "https://app.ogym.fitness/api/workouts/complete") else { return }
        
        var request = AuthViewModel.createAuthenticatedRequest(url: url)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        
        let body: [String: Any] = ["workoutItemId": workoutItemId]
        request.httpBody = try? JSONSerialization.data(withJSONObject: body)
        
        do {
            let (_, response) = try await URLSession.shared.data(for: request)
            if let httpResponse = response as? HTTPURLResponse {
                print("Complete exercise \(workoutItemId) response: \(httpResponse.statusCode)")
            }
        } catch {
            print("Error completing exercise: \(error)")
            // Revert on error
            await MainActor.run {
                completedExercises.remove(workoutItemId)
            }
        }
    }
    
    // MARK: - API: Mark Day as Done
    private func markDayDoneAPI() async {
        let dateFormatter = DateFormatter()
        dateFormatter.dateFormat = "yyyy-MM-dd"
        let today = dateFormatter.string(from: Date())
        
        guard let url = URL(string: "https://app.ogym.fitness/api/member/workout/day/\(today)/mark-done") else { return }
        
        var request = AuthViewModel.createAuthenticatedRequest(url: url)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        
        do {
            let (_, response) = try await URLSession.shared.data(for: request)
            if let httpResponse = response as? HTTPURLResponse {
                print("Mark day done response: \(httpResponse.statusCode)")
                if httpResponse.statusCode == 200 {
                    // Refresh data to show updated state
                    await MainActor.run {
                        onRefresh()
                    }
                }
            }
        } catch {
            print("Error marking day done: \(error)")
        }
    }
}
// MARK: - Exercise Row with Checkbox

    let exercise: WorkoutItem
    let isCompleted: Bool
    let onToggle: () -> Void
    
    var body: some View {
        Button(action: onToggle) {
            HStack(spacing: 12) {
                // Checkbox
                ZStack {
                    Circle()
                        .stroke(isCompleted ? Color.green : Color.gray.opacity(0.5), lineWidth: 2)
                        .frame(width: 28, height: 28)
                    if isCompleted {
                        Circle()
                            .fill(Color.green)
                            .frame(width: 28, height: 28)
                        Image(systemName: "checkmark")
                            .font(.caption.bold())
                            .foregroundColor(.white)
                    }
                }
                
                // Exercise icon
                Image(systemName: "figure.strengthtraining.traditional")
                    .foregroundColor(.indigo)
                
                // Exercise details

                    Text(exercise.exerciseName)
                        .font(.subheadline)

// MARK: - Workout Calendar
struct WorkoutCalendar: View {
    @Binding var currentMonth: Date
    @Binding var selectedDate: Date?
    @Binding var showDayDetail: Bool
    let workoutSummary: WorkoutSummary?
    
    private let calendar = Calendar.current
    private let weekdays = ["Su", "Mo", "Tu", "We", "Th", "Fr", "Sa"]
    
    var body: some View {
        VStack(alignment: .leading, spacing: 12) {
            // Header
            HStack {
                Image(systemName: "calendar")
                    .foregroundColor(.indigo)
                Text("Workout Calendar")
                    .font(.headline)
                
                Spacer()
                
                // Month navigation
                Button(action: previousMonth) {
                    Image(systemName: "chevron.left")
                        .foregroundColor(.secondary)
                }
                
                Text(monthYearString)
                    .font(.subheadline)
                
                Button(action: nextMonth) {
                    Image(systemName: "chevron.right")
                        .foregroundColor(.secondary)
                }
            }
            .padding(.horizontal)
            
            // Legend
            HStack(spacing: 16) {
                LegendItem(color: .green, label: "Present")
                LegendItem(color: .red, label: "Missed")
                LegendItem(color: .blue, label: "Rest")
            }
            .padding(.horizontal)
            
            // Calendar Grid
            VStack(spacing: 8) {
                // Weekday headers
                HStack {
                    ForEach(weekdays, id: \.self) { day in
                        Text(day)
                            .font(.caption2)
                            .foregroundColor(.secondary)
                            .frame(maxWidth: .infinity)
                    }
                }
                
                // Days grid
                let days = daysInMonth()
                LazyVGrid(columns: Array(repeating: GridItem(.flexible()), count: 7), spacing: 8) {
                    ForEach(days, id: \.self) { date in
                        if let date = date {
                            CalendarDayCell(
                                date: date,
                                status: dayStatus(for: date),
                                isToday: calendar.isDateInToday(date),
                                onTap: {
                                    selectedDate = date
                                    showDayDetail = true
                                }
                            )
                        } else {
                            Text("")
                                .frame(width: 36, height: 36)
                        }
                    }
                }
            }
            .padding()
            .background(Color(.systemGray6))
            .cornerRadius(12)
            .padding(.horizontal)
        }
    }
    
    private var monthYearString: String {
        let formatter = DateFormatter()
        formatter.dateFormat = "MMMM yyyy"
        return formatter.string(from: currentMonth)
    }
    
    private func previousMonth() {
        currentMonth = calendar.date(byAdding: .month, value: -1, to: currentMonth) ?? currentMonth
    }
    
    private func nextMonth() {
        currentMonth = calendar.date(byAdding: .month, value: 1, to: currentMonth) ?? currentMonth
    }
    
    private func daysInMonth() -> [Date?] {
        guard let range = calendar.range(of: .day, in: .month, for: currentMonth),
              let firstDay = calendar.date(from: calendar.dateComponents([.year, .month], from: currentMonth)) else {
            return []
        }
        
        let firstWeekday = calendar.component(.weekday, from: firstDay)
        var days: [Date?] = Array(repeating: nil, count: firstWeekday - 1)
        
        for day in range {
            if let date = calendar.date(byAdding: .day, value: day - 1, to: firstDay) {
                days.append(date)
            }
        }
        
        return days
    }
    
    private func dayStatus(for date: Date) -> DayStatus {
        // TODO: Fetch actual workout history from API
        let today = Date()
        if calendar.isDateInToday(date) {
            return .today
        } else if date > today {
            return .future
        } else {
            // Simulate based on streak
            let daysAgo = calendar.dateComponents([.day], from: date, to: today).day ?? 0
            let streak = workoutSummary?.streak ?? 0
            if daysAgo <= streak && daysAgo > 0 {
                return .present
            }
            return .none
        }
    }
}

enum DayStatus {
    case present, missed, rest, today, future, none
}

// MARK: - This Week Progress
struct ThisWeekProgress: View {
    let workoutSummary: WorkoutSummary?
    
    private let weekdays = ["M", "T", "W", "T", "F", "S", "S"]
    
    var body: some View {
        VStack(alignment: .leading, spacing: 12) {
            HStack {
                Text("This Week")
                    .font(.headline)
                Spacer()
                Text("\(completedDays)/3 days")
                    .font(.caption)
                    .foregroundColor(.secondary)
            }
            .padding(.horizontal)
            
            HStack(spacing: 8) {
                ForEach(0..<7, id: \.self) { index in
                    VStack(spacing: 6) {
                        ZStack {
                            RoundedRectangle(cornerRadius: 8)
                                .fill(dayColor(for: index))
                                .frame(width: 40, height: 40)
                            
                            if isCompleted(index) {
                                Image(systemName: "checkmark")
                                    .foregroundColor(.white)
                                    .font(.system(size: 16, weight: .bold))
                            } else if isToday(index) {
                                RoundedRectangle(cornerRadius: 6)
                                    .stroke(Color.indigo, lineWidth: 2)
                                    .frame(width: 36, height: 36)
                            }
                        }
                        
                        Text(weekdays[index])
                            .font(.caption2)
                            .foregroundColor(.secondary)
                    }
                }
            }
            .frame(maxWidth: .infinity)
            .padding()
            .background(Color(.systemGray6))
            .cornerRadius(12)
            .padding(.horizontal)
        }
    }
    
    private var completedDays: Int {
        workoutSummary?.last7DaysCount ?? 0
    }
    
    private func isCompleted(_ index: Int) -> Bool {
        let calendar = Calendar.current
        let today = calendar.component(.weekday, from: Date())
        let mondayBasedToday = (today + 5) % 7
        let streak = workoutSummary?.streak ?? 0
        if index < mondayBasedToday {
            return index >= (mondayBasedToday - streak)
        }
        return false
    }
    
    private func isToday(_ index: Int) -> Bool {
        let calendar = Calendar.current
        let today = calendar.component(.weekday, from: Date())
        let mondayBasedToday = (today + 5) % 7
        return index == mondayBasedToday
    }
    
    private func dayColor(for index: Int) -> Color {
        if isCompleted(index) {
            return .green
        } else if isToday(index) {
            return .indigo.opacity(0.2)
        } else {
            return .clear
        }
    }
}
