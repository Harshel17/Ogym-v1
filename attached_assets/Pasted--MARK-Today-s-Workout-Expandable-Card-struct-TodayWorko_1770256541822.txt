// MARK: - Today's Workout Expandable Card
struct TodayWorkoutExpandable: View {
    let todayWorkout: TodayWorkout?
    @Binding var isExpanded: Bool
    @Binding var completedExercises: Set<Int>
    let onStartWorkout: () -> Void
    
    var exercises: [WorkoutItem] {
        todayWorkout?.items ?? []
    }
    
    var completedCount: Int {
        completedExercises.count
    }
    
    var totalCount: Int {
        exercises.count
    }
    
    var isAllCompleted: Bool {
        completedCount == totalCount && totalCount > 0
    }
    
    var body: some View {
        VStack(spacing: 0) {
            // Header
            Button(action: { withAnimation { isExpanded.toggle() } }) {
                HStack(spacing: 12) {
                    // Dumbbell icon
                    ZStack {
                        RoundedRectangle(cornerRadius: 12)
                            .fill(Color.indigo)
                            .frame(width: 44, height: 44)
                        Image(systemName: "dumbbell.fill")
                            .foregroundColor(.white)
                    }
                    
                    VStack(alignment: .leading, spacing: 2) {
                        Text("Today's Workout")
                            .font(.subheadline)
                            .foregroundColor(.secondary)
                        if let workout = todayWorkout, let label = workout.dayLabel {
                            Text(label)
                                .font(.headline)
                        } else {
                            Text("No workout scheduled")
                                .font(.headline)
                        }
                    }
                    
                    Spacer()
                    
                    if !exercises.isEmpty {
                        // Done button
                        Button(action: markAllDone) {
                            Text(isAllCompleted ? "Done" : "Done")
                                .font(.caption.bold())
                                .padding(.horizontal, 12)
                                .padding(.vertical, 6)
                                .background(isAllCompleted ? Color.green : Color.green)
                                .foregroundColor(.white)
                                .cornerRadius(16)
                        }
                        
                        Image(systemName: isExpanded ? "chevron.up" : "chevron.down")
                            .foregroundColor(.secondary)
                    }
                }
                .padding()
