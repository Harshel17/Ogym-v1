import Foundation

// MARK: - Workout Cycle (The main workout plan)
struct WorkoutCycle: Codable, Identifiable {
    let id: Int
    let name: String
    let gymId: Int?
    let trainerId: Int?
    let memberId: Int
    let cycleLength: Int
    let dayLabels: [String]?
    let restDays: [Int]?
    let startDate: String
    let endDate: String
    let progressionMode: String?
    let currentDayIndex: Int?
    let lastWorkoutDate: String?
    let isActive: Bool?
    let source: String?
    let phaseNumber: Int?
    let createdAt: String?
}

// MARK: - Workout Item (Exercise in a cycle)
struct WorkoutItem: Codable, Identifiable {
    let id: Int
    let cycleId: Int?
    let dayIndex: Int?
    let muscleType: String?
    let bodyPart: String?
    let exerciseName: String
    let sets: Int?
    let reps: Int?
    let weight: String?
    let restSeconds: Int?
    let notes: String?
    let orderIndex: Int?
    let Completed: Bool?
}

// MARK: - Workout Session
struct WorkoutSession: Codable, Identifiable {
    let id: Int
    let gymId: Int?
    let memberId: Int
    let cycleId: Int?
    let date: String
    let dayIndex: Int?
    let status: String?
    let notes: String?
    let createdAt: String?
}

// MARK: - Session Exercise (Completed exercise in a session)
struct SessionExercise: Codable, Identifiable {
    let id: Int
    let sessionId: Int
    let workoutItemId: Int?
    let exerciseName: String
    let setsCompleted: Int?
    let repsCompleted: Int?
    let weightUsed: String?
    let notes: String?
}

// MARK: - Training Phase (for gym members with trainer-led workouts)
struct TrainingPhase: Identifiable, Codable {
    let id: Int
    let gymId: Int
    let memberId: Int
    let trainerId: Int?
    let name: String
    let goalType: String
    let startDate: String
    let endDate: String
    let cycleId: Int?
    let dietPlanId: Int?
    let autoAssignCycle: Bool?
    let useCustomExercises: Bool?
    let cycleLength: Int?
    let dayLabels: [String]?
    let restDays: [Int]?
    let notes: String?
    let createdAt: String?
}

// MARK: - Member Workout Schedule (for gym members)
struct MemberWorkoutSchedule: Codable {
    let cycleId: Int?
    let cycleName: String?
    let cycleLength: Int
    let startDate: String?
    let endDate: String?
    let schedule: [ScheduleDay]?
    
    struct ScheduleDay: Codable, Identifiable {
        var id: String { date }
        let date: String
        let dayIndex: Int
        let dayLabel: String
        let status: String
        let isManuallyCompleted: Bool?
        let completedExercises: Int?
        let totalExercises: Int?
        let sessionId: Int?
    }
}

// MARK: - Today's Workout Response
struct TodayWorkout: Codable {
    let cycleId: Int?
    let cycleName: String?
    let dayIndex: Int?
    let dayLabel: String?
    let isRestDay: Bool?
    let items: [WorkoutItem]?
    let dayManuallyCompleted: Bool?
    let message: String?
}

// MARK: - Workout Summary (Stats)
struct WorkoutSummary: Codable {
    let streak: Int
    let totalWorkouts: Int
    let last7DaysCount: Int
    let thisMonthCount: Int
}

// MARK: - Complete Exercise Request
struct CompleteExerciseRequest: Codable {
    let workoutItemId: Int
    let actualSets: Int
    let actualReps: Int
    let actualWeight: String?
}
