// MARK: - Home Tab (Role-Based)
struct HomeTab: View {
    @EnvironmentObject var authViewModel: AuthViewModel
    
    var body: some View {
        NavigationStack {
            ZStack {
                Color(red: 15/255, green: 23/255, blue: 42/255)
                    .ignoresSafeArea()
                
                ScrollView {
                    VStack(spacing: 20) {
                        // Welcome Header
                        HStack {
                            VStack(alignment: .leading) {
                                Text("Welcome back,")
                                    .foregroundColor(.gray)
                                Text(authViewModel.user?.username ?? "User")
                                    .font(.title2)
                                    .fontWeight(.bold)
                                    .foregroundColor(.white)
                            }
                            Spacer()
                            
                            // Role Badge
                            if let role = authViewModel.user?.role {
                                Text(role.rawValue.capitalized)
                                    .font(.caption)
                                    .padding(.horizontal, 12)
                                    .padding(.vertical, 6)
                                    .background(roleColor(role).opacity(0.3))
                                    .foregroundColor(roleColor(role))
                                    .cornerRadius(20)
                            }
                        }
                        .padding()
                        
                        // Role-Specific Content
                        switch authViewModel.user?.role {
                        case .owner:
                            OwnerDashboard()
                        case .trainer:
                            TrainerDashboard()
                        case .member:
                            MemberDashboard()
                        default:
                            MemberDashboard()
                        }
                        
                        Spacer(minLength: 100)
                    }
                }
            }
            .navigationBarHidden(true)
        }
    }
    
    func roleColor(_ role: UserRole) -> Color {
        switch role {
        case .owner: return .purple
        case .trainer: return .orange
        case .member: return .indigo
        }
    }
}

// MARK: - Workouts Tab
struct WorkoutsTab: View {
    @EnvironmentObject var authViewModel: AuthViewModel
    @State private var workoutCycles: [WorkoutCycle] = []
    @State private var trainingPhase: TrainingPhase? = nil
    @State private var isLoading = true
    @State private var selectedDay = 0
    
    var body: some View {
        NavigationStack {
            ZStack {
                Color(red: 15/255, green: 23/255, blue: 46/255)
                    .ignoresSafeArea()
                
                VStack(spacing: 0) {
                    // Header
                    HStack {
                        Text("Workouts")
                            .font(.largeTitle)
                            .fontWeight(.bold)
                            .foregroundColor(.white)
                        Spacer()
                        
                        Button(action: {}) {
                            Image(systemName: "plus.circle.fill")
                                .font(.title2)
                                .foregroundColor(.indigo)
                        }
                    }
                    .padding()
                    
                    if isLoading {
                        Spacer()
                        ProgressView()
                            .progressViewStyle(CircularProgressViewStyle(tint: .white))
                        Spacer()
                    } else if let phase = trainingPhase {
                        // Gym member - show training phase
                        ScrollView {
                            VStack(spacing: 16) {
                                TrainingPhaseCard(phase: phase)
                            }
                            .padding()
                        }
                    } else if !workoutCycles.isEmpty {
                        // Personal mode - show cycles
                        ScrollView {
                            VStack(spacing: 16) {
                                ForEach(workoutCycles) { cycle in
                                    WorkoutCycleCard(cycle: cycle)
                                }
                            }
                            .padding()
                        }
                    } else {
                        // Empty State
                        Spacer()
                        VStack(spacing: 16) {
                            Image(systemName: "dumbbell.fill")
                                .font(.system(size: 60))
                                .foregroundColor(.gray.opacity(0.5))
                            
                            Text("No Workout Plans")
                                .font(.title2)
                                .fontWeight(.semibold)
                                .foregroundColor(.white)
                            
                            Text("Create your first workout cycle to get started")
                                .font(.subheadline)
                                .foregroundColor(.gray)
                                .multilineTextAlignment(.center)
                            
                            Button(action: {}) {
                                HStack {
                                    Image(systemName: "plus")
                                    Text("Create Workout Plan")
                                }
                                .foregroundColor(.white)
                                .padding()
                                .background(Color.indigo)
                                .cornerRadius(12)
                            }
                            .padding(.top, 8)
                        }
                        .padding()
                        Spacer()
                    }
                }
            }
            .navigationBarHidden(true)
            .onAppear {
                fetchWorkoutCycles()
            }
        }
    }
    
    func fetchWorkoutCycles() {
        let user = authViewModel.user
        let isPersonalOrSelfGuided = user?.gymId == nil || user?.trainingMode == .selfGuided
        
        if isPersonalOrSelfGuided {
            // Personal mode - fetch cycles
            guard let url = URL(string: "https://app.ogym.fitness/api/personal/cycles") else {
                isLoading = false
                return
            }
            let request = AuthViewModel.createAuthenticatedRequest(url: url)
            
            URLSession.shared.dataTask(with: request) { data, response, error in
                DispatchQueue.main.async {
                    isLoading = false
                    guard let data = data else { return }
                    if let cycles = try? JSONDecoder().decode([WorkoutCycle].self, from: data) {
                        self.workoutCycles = cycles
                        print("Loaded \(cycles.count) personal cycles")
                    }
                }
            }.resume()
        } else {
            // Gym member - fetch training phase
            guard let url = URL(string: "https://app.ogym.fitness/api/member/active-phase") else {
                isLoading = false
                return
            }
            let request = AuthViewModel.createAuthenticatedRequest(url: url)
            
            URLSession.shared.dataTask(with: request) { data, response, error in
                DispatchQueue.main.async {
                    isLoading = false
                    guard let data = data else { return }
                    
                    // Debug
                    if let json = String(data: data, encoding: .utf8) {
                        print("Phase response: \(json)")
                    }
                    
                    if let phase = try? JSONDecoder().decode(TrainingPhase.self, from: data) {
                        self.trainingPhase = phase
                        print("Loaded phase: \(phase.name)")
                    }
                }
            }.resume()
        }
    }
}

            HStack {
                Text(phase.name)
                    .font(.headline)
                    .foregroundColor(.white)
                Spacer()

                    .font(.caption)
                    .padding(.horizontal, 8)
                    .padding(.vertical, 4)
                    .background(Color.indigo.opacity(0.3))
                    .cornerRadius(8)
                    .foregroundColor(.indigo)
            }
            
            HStack(spacing: 16) {
                Label("\(phase.cycleLength ?? 3) day cycle", systemImage: "calendar")
                Spacer()
            }
            .font(.caption)
            .foregroundColor(.gray)
            
            // Date range
            HStack {
                Label(phase.startDate, systemImage: "clock")
                Text("→")
                Text(phase.endDate)
            }
            .font(.caption)
            .foregroundColor(.gray)
            
            if let labels = phase.dayLabels, !labels.isEmpty {
                ScrollView(.horizontal, showsIndicators: false) {
                    HStack(spacing: 8) {
                        ForEach(labels.indices, id: \.self) { index in
                            Text(labels[index])
                                .font(.caption2)
                                .padding(.horizontal, 8)
                                .padding(.vertical, 4)
                                .background(Color.gray.opacity(0.2))
                                .cornerRadius(6)
                                .foregroundColor(.white)
                        }
                    }
                }
            }
        }
        .padding()
        .background(Color.gray.opacity(0.15))
        .cornerRadius(16)
    }
}
